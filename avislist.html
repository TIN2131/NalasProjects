🔧 Quick Setup

Connect to Google Sheets for real-time sync between you and Ingrid!

Quick Steps:
1. Create a Google Sheet
2. Add these columns: Category, Item, Priority, Completed, Timestamp, AddedBy, Price, Link, Notes
3. Go to Extensions → Apps Script
4. Copy the code below
5. Deploy as Web App → Anyone can access
📋 Google Apps Script Code 
function doGet() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  
  // Skip header row and return data
  var items = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][0]) { // If category exists
      items.push({
        category: data[i][0],
        item: data[i][1],
        priority: data[i][2],
        completed: data[i][3] === true || data[i][3] === 'true',
        timestamp: data[i][4],
        addedBy: data[i][5],
        price: data[i][6] || 0,
        link: data[i][7] || '',
        notes: data[i][8] || ''
      });
    }
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({items: items}))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader('Access-Control-Allow-Origin', '*');
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  if (data.action === 'add') {
    sheet.appendRow([
      data.category,
      data.item,
      data.priority,
      data.completed,
      data.timestamp,
      data.addedBy,
      data.price || '',
      data.link || '',
      data.notes || ''
    ]);
  } else if (data.action === 'update') {
    var dataRange = sheet.getDataRange();
    var values = dataRange.getValues();
    
    for (var i = 1; i < values.length; i++) {
      if (values[i][1] === data.originalItem && values[i][0] === data.category) {
        sheet.getRange(i + 1, 1, 1, 9).setValues([[
          data.category,
          data.item,
          data.priority,
          data.completed,
          data.timestamp,
          data.addedBy,
          data.price || '',
          data.link || '',
          data.notes || ''
        ]]);
        break;
      }
    }
  } else if (data.action === 'delete') {
    var dataRange = sheet.getDataRange();
    var values = dataRange.getValues();
    
    for (var i = values.length - 1; i >= 1; i--) {
      if (values[i][1] === data.item && values[i][0] === data.category) {
        sheet.deleteRow(i + 1);
        break;
      }
    }
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({status: 'success'}))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader('Access-Control-Allow-Origin', '*');
}
